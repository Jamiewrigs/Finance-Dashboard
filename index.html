      const feed = document.getElementById('news-feed');

      feed.innerHTML = '';

        const li = document.createElement('li');

        li.innerHTML = `<a href="${item.link}" target="_blank">${item.title}</a>`;

        feed.appendChild(li);

      });

    } catch (e) {
